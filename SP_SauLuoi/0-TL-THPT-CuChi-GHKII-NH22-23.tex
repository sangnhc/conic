\begin{bt}%[0K6BH-2]
	Giải các phương trình sau:
	\begin{listEX}[2]
		\item $\sqrt{x^2+x-2}=x-4$.
		\item $\sqrt{x^2-4}=\sqrt{2x^2+x-16}$.
	\end{listEX}
\loigiai{
	\begin{enumerate}
		\item Bình phương hai vế phương trình ta được:
		\begin{eqnarray*}
			&&x^2+x-2=(x-4)^2\\
			&\Leftrightarrow&x^2+x-2=x^2-8x+16\\
			&\Leftrightarrow&9x=18\\
			&\Leftrightarrow&x=2.
		\end{eqnarray*}
	Thử lại, ta thay giá trị $x=2$ vào phương trình đã cho ta thấy không thỏa mãn. \\
	Vậy phương trình đã cho vô nghiệm.
	\item Bình phương hai vế phương trình ta được:
	\begin{eqnarray*}
		&&x^2-4=2x^2+x-16\\
		&\Leftrightarrow&x^2+x-12=0\\
		&\Leftrightarrow&\hoac{&x=3\\&x=-4.}
	\end{eqnarray*}
Thử lại, ta thay các giá trị $x=3$, $x=-4$ vào phương trình đã cho, ta thấy cả hai giá trị đều thỏa mãn.\\
Vậy phương trình đã cho có nghiệm $x=3$, $x=-4$.
	\end{enumerate}
}
\end{bt}

\begin{bt}%[0K6GF-5]
	Một quả bóng được ném thẳng lên từ mặt đất. Khoảng cách của bóng so với mặt đất sau $t$ giấy được cho bởi hàm số $h(t)=at^2+bt$ ($a<0$) với $h(t)$ tính bằng đơn vị mét.
	\begin{enumerate}
		\item Người ta thấy rằng tại thời điểm $t=3{,}75$ giây thì quả bóng đạt được độ cao lớn nhất là $28{,}125$ mét. Tìm $a$ và $b$.
		\item Hỏi quả bóng nằm ở độ cao trên $13$ mét trong thời gian bao lâu?
	\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item Theo đề, ta có hệ phương trình
		\begin{eqnarray*}
		&				&	\heva{	&h(3{,}75)=28{,}125\\
							&-\dfrac{b}{2a}=3{,}75}\\
		&\Leftrightarrow&	\heva{	&a\cdot(3{,}73)^2+b\cdot 3{,}75=28{,}125\\
									&-b=2\cdot 3{,}75}\\
		&\Leftrightarrow&	\heva{	&14{,}0625a+3{,75}b=28{,}125\\
									&7{,}5a+b=0}\\
		&\Leftrightarrow&	\heva{	&a=-2\\
									&b=15.}
		\end{eqnarray*}
	\item Ta có $h(t)=-2t^2+15t$.\\
	Quả bóng nằm ở độ cao trên $13$ mét khi và chỉ khi
	\begin{eqnarray*}
		&&-2t^2+15t> 13\\
		&\Leftrightarrow&-2t^2+15t-13> 0\\
		&\Leftrightarrow&1< t< \dfrac{13}{2}.
	\end{eqnarray*}
	Vậy quả bóng nằm ở độ cao trên $13$ mét trong khoảng thời gian nhỏ hơn $5{,}5$ giây.
	\end{enumerate}
}
\end{bt}

\item Lập phương trình tổng quát của đường thẳng $BC$.
\item Lập phương trình đường tròn $(T)$ ngoại tiếp $\triangle ABC$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Vì $\vec{BC}= (-6;4)$ là VTCP của $BC$ nên $\vec{n_{BC}} = (4;6)$ là VTPT của $BC$. \\
