\begin{bt}%[0T7Y2-1]%[Dự án đề kiểm tra HKII NH22-23- Nguyễn Cường]%[THPT Lương Văn Can]
Giải bất phương trình: $-3x^2+4x-1\geq 0$.
\loigiai{
Đặt $f(x)=-3x^2+4x-1$.\\
Cho $f(x)=0\Leftrightarrow -3x^2+4x-1=0\Leftrightarrow\hoac{&x=1\\&x=\dfrac{1}{3}.}$\\
Bảng xét dấu
\begin{center}
	\begin{tikzpicture}
		\tkzTabInit[nocadre=false,lgt=2,espcl=3,deltacl=0.5]{$x$/1 ,$f(x)$/1}
		{$-\infty$ , $\dfrac{1}{3}$ , $1$ , $+\infty$}
		\tkzTabLine{ , - , 0 , + , 0 , - ,  }
	\end{tikzpicture}
\end{center}
Dựa vào bảng xét dấu, bất phương trình có nghiệm là $\dfrac{1}{3}\leq x\leq 1$.\\
Vậy bất phương trình có nghiệm $\dfrac{1}{3}\leq x\leq 1$.
}
\end{bt}

%%=====Bài 2
\begin{bt}%[0T7B3-2]%[Dự án đề kiểm tra HKII NH22-23- Nguyễn Cường]%[THPT Lương Văn Can]
Giải phương trình: $\sqrt{2x^2-5x+2}=x-2$.
\loigiai{
Điều kiện xác định: $2x^2-5x+2\geq 0\Leftrightarrow \hoac{&x\leq \dfrac{1}{2}\\&x\geq 2.}$\\
Bình phương hai vế của phương trình, ta được
\allowdisplaybreaks
\begin{eqnarray*}
2x^2-5x+2=(x-2)^2&\Leftrightarrow &2x^2-5x+2=x^2-4x+4\\
&\Leftrightarrow &x^2-x-2=0\\
&\Leftrightarrow &\hoac{&x=-1\\&x=2.}
\end{eqnarray*}
Thế $x=-1$ và $x=2$ vào phương trình ban đầu thì nhận thấy chỉ có $x=2$ là thỏa mãn.\\
Vậy tập nghiệm của phương trình là $S=\{2\}$.
}
\end{bt}

%%=====Bài 3
\begin{bt}%[0T8B2-1]%[Dự án đề kiểm tra HKII NH22-23- Nguyễn Cường]%[THPT Lương Văn Can]
Tổ 1 của lớp $10A$ gồm có $6$ học sinh nam và $4$ học sinh nữ.
\begin{enumerate}
	\item Có bao nhiêu cách sắp xếp tất cả học sinh của Tổ 1 thành 1 hàng dọc?
	\item Có bao nhiêu cách chọn 2 học sinh của Tổ 1 làm công việc trực nhật lớp sao cho trong 2 học sinh được chọn có đủ học sinh nam và học sinh nữ?
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Việc sắp xếp tất cả học sinh của Tổ 1 thành 1 hàng dọc là hoán vị của $10$ phần tử.\\
	Vậy có $10!=3\,628\,800$ (cách).
	\item Ta thực hiện việc chọn theo 2 bước
	\begin{itemize}
		\item \textit{Bước 1:} Chọn 1 học sinh nam trong $6$ học sinh nam: có $\mathrm{C}_6^1=6$ cách.
		\item \textit{Bước 2:} Chọn 1 học sinh nữ trong $4$ học sinh nữ: có $\mathrm{C}_4^1=4$ cách.
	\end{itemize}
	Theo quy tắc nhân, ta có $6\cdot 4=24$ cách chọn.\\
	Vậy có $24$ cách chọn 2 học sinh của Tổ 1 làm công việc trực nhật lớp sao cho trong 2 học sinh được chọn có đủ học sinh nam và học sinh nữ.
\end{enumerate}
}
\end{bt}

\begin{bt}%[0T8B2-2]%[Dự án đề kiểm tra HKII NH22-23- Thầy Hóa]%[THPT Lương Văn Can]
Cho tập hợp $X=\{0;1;2;3;5;6;7;8;9\}$. Từ tập $X$ có thể lập được bao nhiêu số tự nhiên có năm chữ số đôi một khác nhau và chia hết cho 2.
\loigiai{
Giả sử số cần tìm có dạng $\overline{abcde}$, $\left(a,b,c,d,e\in X;a\neq 0\right)$.
\begin{itemize}
	\item \textit{Trường hợp 1:} Nếu $e=0$: có 1 cách chọn $e$.\\
	Việc chọn các số $a,b,c,d$ từ tập $X$ là một chỉnh hợp chập $4$ của $8$ phần tử: có $\mathrm{A}_8^4=1680$ (cách chọn).\\
	Theo quy tắc nhân, ta có $1\cdot 1680=1680$ số.
	\item \textit{Trường hợp 2:} Nếu $e\neq 0$: có 3 cách chọn $e$ từ tập $X$.\\
	Chọn $a$ từ tập $X$ có $7$ cách (do $a\neq 0$, $a\neq e$).\\
	Việc chọn các số $b,c,d$ từ tập $X$ là một chỉnh hợp chập $3$ của $7$ phần tử: có $\mathrm{A}_7^3=210$ (cách chọn).\\
	Theo quy tắc nhân, ta có $3\cdot 7\cdot 210 =4410$ số.
\end{itemize}
	Theo quy tắc cộng, ta có $1680+4410=6090$ (số cần tìm).
}
\end{bt}

Khai triển biểu thức sau: $(x-2y)^5$.
\loigiai{
Áp dụng khai triển nhị thức NewTon, ta được
\allowdisplaybreaks
\begin{eqnarray*}
(x-2y)^5&=&\mathrm{C}_5^0\cdot x^5+\mathrm{C}_5^1\cdot x^4\cdot (-2y)+\mathrm{C}_5^2\cdot x^3\cdot (-2y)^2+\mathrm{C}_5^3\cdot x^2\cdot (-2y)^3\\
&&+\mathrm{C}_5^4\cdot x\cdot (-2y)^4+\mathrm{C}_5^5\cdot (-2y)^5\\
&=&x^5-10x^4y+40x^3y^2-80x^2y^3+80xy^4-32y^5.
\end{eqnarray*}
}
\end{bt}

\begin{bt}%[0T9K3-3]%[Dự án đề kiểm tra HKII NH22-23- Thầy Hóa]%[THPT Lương Văn Can]
Trong mặt phẳng tọa độ $Oxy$, cho ba điểm $A(1;3)$, $B(5;1)$, $C(-1;1)$.
\begin{enumerate}
	\item Tìm tọa độ điểm $D$ nằm trên trục hoành sao cho tam giác $ABD$ cân tại $D$.
	\item Viết phương trình đường thẳng $\Delta$ qua $C$ và vuông góc với đường thẳng
	\[d:2x-y-1=0. \]
	\item Viết phương trình đường tròn $(C)$ có tâm $I(-1;2)$ và tiếp xúc với đường thẳng
	\[b:3x-4y+1=0. \]
	\item Viết phương trình tiếp tuyến của đường tròn $(C)$ tại điểm $E(-3;2)$.
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Vì $D\in Ox$ nên $D\left(x;0\right)$.\\
	Do tam giác $ABD$ cân tại $D$ nên
	\allowdisplaybreaks
	\begin{eqnarray*}
	AD=BD&\Leftrightarrow&\left|\vec{AD}\right|=\left|\vec{BD}\right|\\
	&\Leftrightarrow&\sqrt{(x-1)^2+(-3)^2}=\sqrt{(x-5)^2+(-1)^2}\\
	&\Leftrightarrow&x^2-2x+1+9=x^2-10x+25+1\\
	&\Leftrightarrow&8x=16\\
	&\Leftrightarrow&x=2.
	\end{eqnarray*}
	Vậy $D(2;0)$.
	\item Vì $\Delta$ vuông góc với $d:2x-y-1=0$ nên $\Delta:x+2y+c=0$.\\
	Do $\Delta$ đi qua $C$ nên $-1+2+c=0\Leftrightarrow c=-1$.\\
	Vậy $\Delta:x+2y-1=0$.
	\item Vì đường $(C)$ tiếp xúc với $b$ nên
	\[R=\mathrm{d}\left(I,b\right)=\dfrac{\left|-3-8+1\right|}{\sqrt{3^2+(-4)^2}}=2. \]
	Phương trình đường tròn $(C):(x+1)^2+(y-2)^2=4$.
	\item Gọi $a$ là phương trình tiếp tuyến của đường tròn $(C)$ tại điểm $E$.\\
	Khi đó $a$ nhận $\vec{IE}=(-2;0)$ làm véc-tơ pháp tuyến.\\
	Phương trình đường thẳng $a$ đi qua $E$ nhận $\vec{IE}$ làm véc-tơ pháp tuyến
	\[-2\cdot (x+3)+0\cdot (y-2)=0\Leftrightarrow x+3=0. \]
	Vậy phương trình tiếp tuyến của đường tròn $(C)$ tại điểm $E$ là $x+3=0$.
\end{enumerate}
}
\end{bt}
