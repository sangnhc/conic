 		Cỡ mẫu là $n=10$.\\
 		Đặt $A=6+6+6+7+8=33$.\\
 		Đặt $B=8+8+9+10+10=45$.\\
 		Số trung bình $\overline{x}=\dfrac{A+B}{10}=7{,}8$.
 		\item Độ lệch chuẩn.\\
 		Đặt $C=1{,}8^2+1{,}8^2+1{,}8^2+0{,}8^2+0{,}2^2=10{,}4$.\\
 		Đặt $D=0{,}2^2+0{,}2^2+1{,}2^2+2{,}2^2+2{,}2^2=11{,}2$.\\
 		Phương sai mẫu là 
 		\[S^2=\dfrac{1}{10}\left(C+D\right)\approx2{,}16.\]
 	\end{itemize}
	 	\item Tìm khoảng tứ phân vị của mẫu số liệu.\\ 
	 	\begin{itemize}
	 		\item Tứ phân vị.\\
	 		Vì cỡ mẫu là số chẵn nên trung vị mẫu là $\dfrac{1}{2}(8+8)=8$.\\
	 		Tứ phân vị thứ hai là trung vị của mẫu số liệu đã cho nên $Q_2=8$.\\
	 		Tứ phân vị thứ nhất là trung vị của mẫu
	 		$$6, 6, 6, 7, 8$$
	 		Do đó $Q_1=6$.\\
	 		Tứ phân vị thứ ba là trung vị của mẫu 
	 		$$8, 8, 9, 10, 10$$
	 		Do đó $Q_3=9$.
	 		\item Khoảng tứ phân vị.\\
	 		Khoảng tứ phân vị là $\Delta_Q=Q_3-Q_1=9-6=3$.
	 	\end{itemize}
	 \end{enumerate}
