\begin{bt}%[0D3B1-2]%[Dự án đề kiểm tra HKI NH22-23 - Phạm Phương]%[THCS&THPT Lương Thế Vinh - Sở Hà Nội]
	Tìm tập xác định của hàm số $y=\sqrt{x-2}+\dfrac{2x+5}{x^2-9}$.
\loigiai{
Hàm số đã cho có nghĩa khi $\heva{&x-2 \geq 0 \\& x^2-9 \neq 0} \Leftrightarrow \heva{& x \geq 2 \\& x \neq 3 \\& x \neq -3} \Leftrightarrow \heva{& x \geq 2 \\& x \neq 3.}$\\

\begin{bt}%[0D3B1-4]%[Dự án đề kiểm tra HKI NH22-23 - Phạm Phương]%[THCS&THPT Lương Thế Vinh - Sở Hà Nội]
	Với giá trị nào của tham số $m$ sao cho đồ thị hàm số $y=x^2+mx+3$ đi qua điểm $A(1;4)$?
	\loigiai{
	Đồ thị hàm số $y=x^2+mx+3$ đi qua điểm $A(1;4)$ nên ta có 
	$1^2+m\cdot 1+3=4 \Leftrightarrow  m=0$.	
}
\end{bt}

	\loigiai{
	Hàm số $y=x^2-4x+5$ có $a=1>0$ và tọa độ đỉnh $S$ gồm 
	$x_S=-\dfrac{b}{2a}=-\dfrac{(-4)}{2\cdot 1}=2$ và $y_S=y(2)=2^2-4\cdot 2+5=1$. Ta có bảng biến thiên sau
	\begin{center}
		\begin{tikzpicture}
			\tkzTabInit[nocadre=false,lgt=1.2,espcl=2.5,deltacl=0.6]
			{$x$ /0.6, $y$ /2}
			{$-\infty$,$2$,$+\infty$}
			\tkzTabVar{+/$+\infty$,-/$1$,+/$+\infty$}
		\end{tikzpicture}
	\end{center}
	Từ bảng biến thiên ta có giá trị nhỏ nhất của hàm số trên đoạn $\left[-1;3\right]$ là $1$ tại $x=2$.\\
	Lại có 
	$\heva{&y(-1)=(-1)^2-4(-1)+5=10 \\& y(3)=3^2-4\cdot 3+5=2.}$\\
	Suy ra giá trị lớn nhất của hàm số trên đoạn $\left[-1;3\right]$ là $10$ tại $x=-1$.
}
\end{bt}

	\loigiai{
	Gọi $y$ $(y>0)$ là chiều dài của hình chữ nhật ta có $2\left(x+y\right)=100 \Leftrightarrow y=50-x$.\\
	Khi đó diện tích hình chữ nhật đã cho là $S=xy=x\left(50-x\right)=-x^2+50x$.

	Cho hàm số $y=f(x)=x^{2023}+ax^{2019}+3$ thỏa mãn $f(2022)=2021$. Tính $f(-2022)$.
	\loigiai{
	Ta có $f(2022)=2021 \Leftrightarrow 2022^{2023}+a\cdot2022^{2019}+3=2021 \Leftrightarrow 2022^{2023}+a\cdot2022^{2019}=2018$.
	Khi đó
	\allowdisplaybreaks{
		\begin{eqnarray*}
			f(-2022)&= & (-2022)^{2023}+a\cdot(-2022)^{2019}+3\\
			&= & -\left(2022^{2023}+a\cdot2022^{2019}\right)+3 \\
			&= & -2018+3=-2015.
	\end{eqnarray*} }
Vậy $f(-2022)=-2015$.
}
