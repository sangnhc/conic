\begin{bt}%[0T7B1-1]%[Dự án đề kiểm tra GHKII NH22-23 - Lê Hùng Thắng]%[Trường Chinh]
Tìm các giá trị của tham số $m$ để $3x^2+2 x+1-m>0$, $\forall x \in \mathbb{R}$.
	\loigiai{
	Ta có 
	\allowdisplaybreaks \vspace*{-0.7cm}
	\begin{eqnarray*} 
	& & 3x^2+2 x+1-m>0,\, \forall x \in \mathbb{R} \\
	& \Leftrightarrow & \heva{&3>0 \\& \Delta'<0} \\
	& \Leftrightarrow & 1-3\cdot (1-m) <0 \\
	& \Leftrightarrow & 3m-2 <0 \\
	& \Leftrightarrow & m< \dfrac{2}{3}.
	\end{eqnarray*}
Vậy $m<\dfrac{2}{3}$ thỏa yêu cầu đề bài.
	}
\end{bt}

\begin{bt}%[0T7B3-1]%[Dự án đề kiểm tra GHKII NH22-23 - Lê Hùng Thắng]%[Trường Chinh]
Giải phương trình $\sqrt{x^2-x-4}=2+x$.
	\loigiai{Ta có	\allowdisplaybreaks \vspace*{-0.7cm}
	\begin{eqnarray*} 
	& & \sqrt{x^2-x-4}=2+x \\
	& \Rightarrow & x^2-x-4 = (2+x)^2\\
	& \Rightarrow & x^2-x-4 = x^2+4x+4\\
	& \Rightarrow & 5x+8 =0 \\
	& \Rightarrow & x = -\dfrac{8}{5}.
	\end{eqnarray*}
Thử lại ta có $x=-\dfrac{8}{5}$ thỏa phương trình.\\
Vậy tập nghiệm của phương trình là $S=\left\{-\dfrac{8}{5}\right\}$.
}
\end{bt}

\begin{bt}%[0T7T2-1]%[Dự án đề kiểm tra GHKII NH22-23 - Lê Hùng Thắng]%[Trường Chinh]
Một quả bóng được ném thẳng lên từ độ cao $1{,}7$ m so với mặt đất với vận tốc $12$ m/s. Độ cao của bóng so với mặt đất (tính bằng mét) sau $t$ giây được cho bởi hàm số $h(t)=-4{,}9t^2+12t+1{,}7$. Hỏi quả bóng ở độ cao trên $6$ m trong khoảng thời gian bao lâu (làm tròn kết quả đến hàng phần trăm)?
	\loigiai{Theo đề bài, độ cao của quả bóng tính theo thời gian $t$ (giây) là $h(t)=-4{,}9t^2+12t+1{,}7$.\\
	Do đó 	\allowdisplaybreaks \vspace*{-0.7cm}
	\begin{eqnarray*} 
		&  & h(t)>6\\
		&  \Leftrightarrow  & -4{,}9t^2+12t+1{,}7>6 \\
		&  \Leftrightarrow  & -4{,}9t^2+12t-4{,}3>0 \\
		&  \Leftrightarrow  & t_1<t< t_2. 
	\end{eqnarray*}	
Với $t_1 \approx 0{,}44$ và $t_2\approx 2{,}01$.\\
Vậy khoảng thời gian quả bóng ở độ cao trên $6$ m là\\
$T= \vert t_2-t_1 \vert \approx  2{,}01 - 0{,}44 = 1{,}57$ (giây).
	}
\end{bt}

\begin{bt}%[0T8K2-1]%[Dự án đề kiểm tra GHKII NH22-23 - Quang Anh]%[Trường Chinh]
Có bao nhiêu số tự nhiên chẵn có $5$ chữ số đôi một khác nhau được lấy từ các chữ số $0,1,2,3,4,5,6,7$?
		\loigiai{
			Gọi $\overline{a_1a_2a_3a_4a_5}$ ($a_1 \ne 0$) là số tự nhiên chẵn có $5$ chữ số đôi một khác nhau được lấy từ tập $S=\{0;1;2;3;4;5;6;7\}$.\\
			Để thiết lập được số tự nhiên chẵn và có $5$ chữ số đôi một khác nhau được lấy từ tập $S=\{0;1;2;3;4;5;6;7\}$ ta chia thành các trường hợp như sau
			\begin{itemize}
				\item[TH1:] Số có dạng $\overline{a_1a_2a_3a_40}$. Khi đó để thiết lập số tự nhiên thỏa yêu cầu ta sẽ chọn $4$ số từ tập $S \setminus \{0\}$ và điền vào các vị trí $a_1a_2a_3a_4$: có $\mathrm{A}_7^4$ (cách). \\
				Do đó trong trường hợp này này có $\mathrm{A}_7^4$ (số).
				\item[TH2:] Số có dạng $\overline{a_1a_2a_3a_4a_5}$ với $a_5 \ne 0$. Khi đó để thiết lập số tự nhiên thỏa yêu cầu ta thực hiện liên tiếp các bước sau
				\begin{itemize}
					\item[B1:] Chọn $1$ số trong tập $\{2;4;6\}$ để điền vào vị trí $a_5$: có $3$ (cách).
					\item[B2:] Chọn $1$ số trong tập $S \setminus \{0;a_5\}$ để điền vào vị trí $a_1$: có $6$ (cách).
					\item[B3:] Chọn $3$ số trong tập $S \setminus \{a_1;a_6\}$ và điền vào $3$ vị trí $a_2a_3a_4$: có $\mathrm{A}^3_6$ (cách).
				\end{itemize}
				Theo quy tắc nhân, trường hợp này ta có $3 \cdot 6 \cdot \mathrm{A}^3_6$ (số).
			\end{itemize}
			Vậy theo quy tắc cộng, ta có $\mathrm{A}_7^4+3 \cdot 6 \cdot \mathrm{A}^3_6=3000$ số thỏa yêu cầu bài toán.
	}
\end{bt}
