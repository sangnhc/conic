\begin{bt}%[0T8B1-2]%[Mui Doan]%THPT Phú Nhuận
	Từ một hộp có $4$ bi đỏ, $5$ bi xanh, $6$ bi vàng. Có bao nhiêu cách lấy ra $4$ bi trong mỗi trường hợp sau
	\begin{enumerate}
		\item Các bi được lấy có đúng $2$ bi xanh, $1$ bi đỏ và $1$ bi vàng.
		\item Các bi được lấy cùng $1$ màu.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Các bi được lấy có đúng $2$ bi xanh, $1$ bi đỏ và $1$ bi vàng.\\
		Chọn $2$ bi xanh có $\mathrm{C}_5^2=10$ cách.\\
		Chọn $1$ bi đỏ có $\mathrm{C}_4^1=4$ cách.\\
		Chọn $1$ bi vàng có $\mathrm{C}_6^1=6$ cách.\\
		Theo quy tắc nhân có $10\cdot 4\cdot 6=240$ cách.\\
		Vậy có $240$ cách chọn $4$ bi trong đó có đúng $2$ bi xanh, $1$ bi đỏ và $1$ bi vàng.
		\item Các bi được lấy cùng $1$ màu.
		\begin{itemize}
			\item Trường hợp $1$: Chọn $4$ bi màu đỏ có $\mathrm{C}_4^4=1$ cách.
			\item Trường hợp $2$: Chọn $4$ bi màu xanh có $\mathrm{C}_5^4=5$ cách.
			\item Trường hợp $3$: Chọn $4$ bi màu vàng có $\mathrm{C}_6^4=15$ cách.
		\end{itemize}
	Theo quy tắc cộng ta có $1+5+15=21$ cách.\\
	Vậy có $18$ cách chọn $4$ bi trong đó các bi được lấy cùng $1$ màu.
	\end{enumerate}
	}
\end{bt}

\begin{bt}%[0T0K2-2]%[Mui Doan]%THPT Phú Nhuận
	Lớp $10A_1$ có $20$ học sinh nam và $26$ học sinh nữ. Lớp $10A_2$ có $16$ học sinh nam và $24$ học sinh nữ. Chọn ngẫu nhiên mỗi lớp $3$ học sinh đi dự hội nghị đoàn trường. Tính xác suất của mỗi biến cố sau: (kết quả làm tròn đến hàng phần nghìn)
	\begin{enumerate}
		\item Trong $6$ học sinh được chọn đều là học sinh nữ.
		\item Trong $6$ học sinh được chọn có cả học sinh nam và học sinh nữ.
	\end{enumerate}
	\loigiai{
	Ta có $n(\Omega)=\mathrm{C}_{46}^3$.
		\begin{enumerate}
			\item Gọi $A$ là biến cố \lq\lq Trong $6$ học sinh được chọn đều là học sinh nữ\rq\rq\\
			Ta có $n(A)=\mathrm{C}_{26}^3$.\\
			$\mathrm{P}(A)=\dfrac{n(A)}{n(\Omega)}=\dfrac{\mathrm{C}_{26}^3}{\mathrm{C}_{46}^3}\approx 0{,}171$.
			\item Gọi $B$ là biến \lq\lq Trong $6$ học sinh được chọn có cả học sinh nam và học sinh nữ\rq\rq.
			\begin{itemize}
				\item Trường hợp $1$: chon $1$ nam $2$ nữ có $\mathrm{C}_{20}^1\cdot \mathrm{C}_{26}^2=6\,500$ cách.
				\item Trường hợp $2$: chon $2$ nam $1$ nữ có $\mathrm{C}_{20}^2\cdot \mathrm{C}_{26}^1=4\,940$ cách.
			\end{itemize}
			Ta có $n(B)=6\,500+4\,940=11\,440$.\\
			$\mathrm{P}(B)=\dfrac{n(B)}{n(\Omega)}=\dfrac{11\,440}{\mathrm{C}_{46}^3}\approx 0{,}754$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[0T7B2-3]%[Mui Doan]%THPT Phú Nhuận
	Cho $f(x)=x^2-2(m-1)x+3m-3$. Tìm tất cả giá trị của $m$ để $f(x)\geq 0,\forall x\in\mathbb{R}$.
	\loigiai{
	Ta có 
	\allowdisplaybreaks
	\begin{eqnarray*}
	&&f(x)\geq 0,\forall x\in\mathbb{R}
	\\
	&\Leftrightarrow&\heva{&a>0\\&\Delta'\leq 0}
	\\
	&\Leftrightarrow&\heva{&1>0\\&(m-1)^2-(3m-3)\leq 0}
	\\
	&\Leftrightarrow& m^2-5m+4\leq 0
	\\
	&\Leftrightarrow& 1\leq m\leq 4.
	\end{eqnarray*}
	Vậy $f(x)\geq 0,\forall x\in\mathbb{R}\Leftrightarrow 1\leq m\leq 4$.
	}
\end{bt}

	Bác Hùng muốn trồng một vườn hoa trên mảnh đất hình chữ nhật và làm hàng rào bao quanh. Bác Hùng chỉ mua được $40$ m hàng rào. Gọi $x>0$ (m) là chiều dài hình chữ nhật.
	\begin{enumerate}
		\item Tính diện tích vườn hoa theo $x$ biết rằng bác Hùng sử dụng hết $40$ m hàng rào.
		\item Bác Hùng muốn diện tích vườn hoa ít nhất $64$ m$^2$. Hỏi chiều dài của vườn hoa nằm trong khoảng nào?
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Chiều rộng của mảnh đất hình chữ nhật là $20-x$ (m).\\
			Diện tích của mảnh đất hình chữ nhật là $x(20-x) = 20x-x^2$ (m$^2$).
			\item Theo yêu cầu, ta có
			\[20x - x^2 \geq 64 \Leftrightarrow -x^2 + 20x -64\geq 0 \Leftrightarrow 4\leq x \leq 16.\]
			Mặt khác, ta có điều kiện $x \geq 20-x \Leftrightarrow x \geq 10$.\\
			Vậy chiều dài $x$ của vườn hoa thỏa mãn $10\leq x \leq 16$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Dự án Tex đề HK2 2022-2023 L1]%[Phạm Văn Long]%[0H9K2-2]
	Trong mặt phẳng tọa độ $Oxy$, cho $A(-2;1)$, $B(2;-3)$ và đường thẳng $(d)\colon 3x + 4y + 7 =0$.
	\begin{enumerate}
		\item Viết phương trình tham số của đường thẳng $(d_1)$ đi qua hai điểm $A$, $B$.
		\item Viết phương trình đường thẳng $(d_2)$ đi qua $A$ và vuông góc với đường thẳng $(d)$.
		\item Tính $\cos $ của góc giữa hai đường thẳng $(d_1)$ và $(d_2)$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có $\overrightarrow{AB}=(4;-4)$.\\
			Ta có $(d_1)$ qua $A(-2;1)$ và có véc-tơ chỉ phương $\overrightarrow{AB}=(4;-4)$ nên có phương trình tham số là
			\[\heva{&x = -2 + 4t\\&y=1 -4t} \quad (t\in \mathbb{R}).\]
			\item Ta có $(d_2)\perp (d)\colon 3x + 4y + 7=0$ nên $(d_2)\colon 4x - 3y+m=0$.\\
			Mà $A(-2;1)\in (d_2)$ nên
			\[4\cdot (-2) -3\cdot 1 + m =0 \Leftrightarrow -11 + m =0 \Leftrightarrow m=11.\]
			Vậy $(d_2)\colon 4x - 3y +11=0$.
			\item Ta có $(d_1)$ có véc-tơ pháp tuyến là $\overrightarrow{n}_1 = (4;4)$, $(d_2)$ có véc-tơ pháp tuyến là $\overrightarrow{n}_2 = (4;-3)$.\\
			Khi đó 
			\[\cos (d_1,d_2)=\dfrac{|\overrightarrow{n}_1 \cdot \overrightarrow{n}_2|}{\left|\overrightarrow{n}_1\right|\cdot \left|\overrightarrow{n}_2\right|}=\dfrac{|4\cdot 4 + 4\cdot (-3)|}{\sqrt{4^2 + 4^2}\cdot \sqrt{4^2 + (-3)^2}}=\dfrac{\sqrt{2}}{10}.\]
		\end{enumerate}
	}
\end{bt}

	Trong mặt phẳng tọa độ $Oxy$.
	\begin{enumerate}
		\item Viết phương trình đường tròn $(C)$ có tâm $I(-1;-2)$ và bán kính $R=\sqrt{10}$.
		\item Viết phương trình tiếp tuyến của đường tròn $(C)$, biết tiếp tuyến song song với đường thẳng $(d)\colon x + 3y - 3=0$
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có $(C)$ có tâm $I(-1;-2)$, $R=\sqrt{10}$ nên có phương trình là
			\allowdisplaybreaks
			\begin{eqnarray*}
			(x-a)^2 + (y-b)^2 = R^2 &\Leftrightarrow& 
			(x+1)^2 + (y+2)^2 = 10. 
			\end{eqnarray*}
			\item Gọi $(\Delta)$ là tiếp tuyến cần tìm.\\
			Vì $(\Delta)\parallel (d)\colon x + 3y -3 =0$ nên $(\Delta)\colon x + 3y + m=0$ ($m\neq -3$).\\
			Vì $(\Delta)$ là tiếp tuyến của $(C)$ nên ta có
			\allowdisplaybreaks
			\begin{eqnarray*}
			\mathrm{d}(I,\Delta) = R  
			&\Leftrightarrow& \dfrac{|1\cdot(-1) + 3\cdot (-2) + m |}{\sqrt{1^2 + 3^2}} =\sqrt{10}\\
			&\Leftrightarrow&  \dfrac{|m-7|}{\sqrt{10}}=\sqrt{10}\\
			&\Leftrightarrow&  |m-7| = 10\\
			&\Leftrightarrow& \hoac{&m-7 = 10\\& m-7 =-10.}\\
			&\Leftrightarrow& \hoac{&m=17\quad (\text{nhận})\\&m=-3.\quad (\text{loại})}
			\end{eqnarray*}
			Vậy phương trình tiếp tuyến cần tìm là $(\Delta)\colon x + 3y +17 =0$.
		\end{enumerate}

	}
\end{bt}
