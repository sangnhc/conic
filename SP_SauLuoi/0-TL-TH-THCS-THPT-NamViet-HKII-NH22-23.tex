	\begin{enumerate}
		\item Viết phương trình đường thẳng $BC$.
		\item Viết phương trình đường thẳng $d$ đi qua $A$ và cách đều $B$ và $C$.
	\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item Phương trình đường thẳng $BC$ là
		\begin{eqnarray*}
			&&\dfrac{x-x_B}{x_C-x_B}=\dfrac{y-y_B}{y_C-y_B}\\
			&\Leftrightarrow&\dfrac{x+3}{-1+3}=\dfrac{y-2}{-2-2}\\
			&\Leftrightarrow&\dfrac{x+3}{2}=\dfrac{y-2}{-4}\\
			&\Leftrightarrow&2x+y+4=0.
		\end{eqnarray*}
	\item Gọi $\vec{n}=(a;b)$ là véc-tơ pháp tuyến của $d$.\\
	Phương trình đường thẳng $d$ qua $A(1;1)$ là
	\[
	a(x-1)+b(y-1)=0\Leftrightarrow ax+by-a-b=0.
	\]
	Ta có $d$ cách đều hai điểm $B$ và $C$ nên
	\begin{eqnarray*}
		&&\mathrm{d}(B,d)=\mathrm{d}(C,d)\\
		&\Leftrightarrow&\dfrac{|-3a+2b-a-b|}{\sqrt{a^2+b^2}}=\dfrac{|-a-2b-a-b|}{\sqrt{a^2+b^2}}\\
		&\Leftrightarrow&|-4a+b|=|-2a-3b|\\
		&\Leftrightarrow&\hoac{&-4a+b=-2a-3b\\&-4a+b=2a+3b}\\
		&\Leftrightarrow&\hoac{&-2a=-4b\\&-6a=2b}\\
		&\Leftrightarrow&\hoac{&a=2b\\&-3a=b.}
	\end{eqnarray*}
\begin{itemize}
	\item Với $a=2b$, cho $b=1$, suy ra $a=2$. \\
	Phương trình đường thẳng $d$ là $2x+y-3=0$.
	\item Với $b=-3a$, chọn $a=1$, suy ra $b=-3$. \\
	Phương trình $d$ là $x-3y+2=0$.
\end{itemize}
Nhận xét: Ta có thể giải bài này bằng cách chia ra hai trường hợp:
\begin{itemize}
	\item TH1. $d$ đi qua $A$ và trung điểm $I$ của $BC$.
