\begin{bt}%[0T7B3-2]%[Dự án đề kiểm tra HKII NH22-23- Mui Doan]%[Nguyễn Khuyến]
	Giải các phương trình sau
	\begin{enumerate}
		\item $\sqrt{4-3x}=\sqrt{x^2-x+1}$.
		\item $\sqrt{3x^2-13x+14}=x-3$.
	\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item $\sqrt{4-3x}=\sqrt{x^2-x+1}$.
		Bình phương hai vế phương trình ta được
		\allowdisplaybreaks
		\begin{eqnarray*}
			&&4-3x=x^2-x+1
			\\
			&\Rightarrow&x^2+2x-3=0
			\\
			&\Rightarrow&\hoac{&x=1\\&x=-3.}
		\end{eqnarray*}
	Thử lại với $x=1$ và $x=-3$ đều thỏa phương trình đã cho.\\
	Vậy tập nghiệm của phương trình là $S=\{-3;1\}$.
		\item $\sqrt{3x^2-13x+14}=x-3$.
		Bình phương hai vế phương trình ta được
		\allowdisplaybreaks
		\begin{eqnarray*}
			&&3x^2-13x+14=(x-3)^2
			\\
			&\Leftrightarrow&3x^2-13x+14=x^2-6x+9
			\\
			&\Leftrightarrow&2x^2-7x+5=0
			\\
			&\Rightarrow&\hoac{&x=1\\&x=\dfrac{5}{2}.}
		\end{eqnarray*}
		Thử lại với $x=1$ và $x=\dfrac{5}{2}$ đều không  thỏa phương trình đã cho.\\
		Vậy  phương trình đã cho vô nghiệm.
	\end{enumerate}
	}
\end{bt}

	Tìm $m$ để biểu thức $f(x)=(m-4)x^2+(2m-8)x+m-5$ luôn âm với mọi $x$.
	\loigiai{
	\begin{itemize}
		\item Xét $m-4=0\Leftrightarrow m=4$.\\
		Khi đó $f(x)=-1<0, \forall x\in\mathbb{R}$.\\
		Nhận $m=4$.
		\item Xét $m-4\neq 0\Leftrightarrow m\neq 4$.\\
		Khi đó
		\allowdisplaybreaks
		\begin{eqnarray*}
			&&f(x)<0,\forall x\in\mathbb{R}
			\\
			&\Leftrightarrow&\heva{&a<0\\&\Delta<0}
			\\
			&\Leftrightarrow&\heva{&m-4<0\\&(2m-8)^2-4(m-4)(m-5)<0}
			\\
			&\Leftrightarrow&\heva{&m<4\\&4m-16<0}
			\\
			&\Leftrightarrow&m<4.
		\end{eqnarray*}
	\end{itemize}
	Vậy $f(x)<0,\forall x\in\mathbb{R}\Leftrightarrow m\leq 4$.	
	}
\end{bt}

	\loigiai{

	\allowdisplaybreaks

		&&H\,\text{là trực tâm tam giác}\, ABC
		\\
		&\Leftrightarrow&\heva{&\overrightarrow{CH}\cdot \overrightarrow{AB}=0\\&\overrightarrow{BH}\cdot \overrightarrow{AC}=0}
		\\
		&\Leftrightarrow&\heva{&(x-2)\cdot 6+(y-6)\cdot 0=0\\&(x-3)\cdot 5+y\cdot 6=0}
		\\
		&\Leftrightarrow&\heva{&x=2\\&-5+6y=0}
		\\
		&\Leftrightarrow&\heva{&x=2\\&y=\dfrac{5}{6}.}
	\end{eqnarray*}
