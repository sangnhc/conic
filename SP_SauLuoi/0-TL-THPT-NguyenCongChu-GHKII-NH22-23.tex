	\begin{enumerate}
		\item $d_1\colon x + y - 3=0$; $d_2\colon 2x + 2y=0$
		\item $d_1\colon - 4x + 6y - 2=0$; $d_2\colon 2x - 3y + 1=0$.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item
		\begin{itemize}
			\item $d_1$ và $d_2$ có  véc-tơ pháp tuyến lần lượt là $\vec{n_1}=(1;1)$ và $\vec{n_2}=(2;2)$.
			\item Ta có $\dfrac{1}{1}=\dfrac{2}{2}$, suy ra $\vec{n_1}$ và $\vec{n_2}$ cùng phương.
			\item  Lấy $M(1,2)\in d_1$, thay tọa độ $M$ vào $d_2$, ta được $2\cdot1+2\cdot2=6\neq0$, suy ra $M\not\in d_2$.
		\end{itemize}
		Vậy $d_1\parallel d_2$.
		\item 
		\begin{itemize}
			\item $d_1$ và $d_2$ có  véc-tơ pháp tuyến lần lượt là $\vec{n_1}=(-4;6)$ và $\vec{n_2}=(2;3)$.
			\item Ta có $\dfrac{-4}{6}\neq\dfrac{2}{3}$, suy ra $\vec{n_1}$ và $\vec{n_2}$ không cùng phương.
		\end{itemize}
		Vậy $d_1$ cắt $d_2$.
	\end{enumerate}
	}

\begin{bt}%[0T9B1-3]%[0T9B1-4]%[Dự án đề kiểm tra GHK1 22-23-Nhật Thiện]%[THPT Nguyễn Công Trứ]
	Cho $\vec{a}=(2;1)$, $\vec{b}=(3;4)$, $\vec{c}=(7;2)$.
	\begin{enumerate}
		\item Tìm $\vec{u}$ sao cho $\vec{u}=2\vec{a}-3\vec{b}+\vec{c}$;
		\item Tìm $m$ và $n$ sao cho $\vec{c}=m\vec{a}+n\vec{b}$.
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Ta có $\vec{u}=(x;y)$. Khi đó
	\begin{eqnarray*}
		& &\vec{u}=2\vec{a}-3\vec{b}+\vec{c}\\
		&\Leftrightarrow&\heva{&x=2\cdot2-3\cdot3+7\\&y=2\cdot 1-3\cdot 4+2}\\
		&\Leftrightarrow& \heva{&x=2\\&y=-8.}
	\end{eqnarray*}
	Vậy $\vec{u}=(2;-8)$.
	\item Ta có
	\begin{eqnarray*}
		& &\vec{c}=m\vec{a}+n\vec{b}\\
		&\Leftrightarrow& \heva{&7=m\cdot 2+n\cdot 3\\&2=m\cdot 1+n\cdot 4}\\
		&\Leftrightarrow& \heva{&m=\dfrac{22}{5}\\&n=-\dfrac{3}{5}.}
	\end{eqnarray*}
	Vậy $m=\dfrac{22}{5}$ và $n=-\dfrac{3}{5}$.
\end{enumerate}
}
\end{bt}

	Lập phương trình đường tròn ngoại tiếp tam giác có tọa độ các đỉnh là $A(0;6)$, $B(7;7)$, $C(8;0)$.
	\loigiai{
	Đường tròn ngoại tiếp tam giác $ABC$ có dạng $x^2+y^2-2ax-2by+c=0$ với $a^2+b^2-c>0$.\\
	Đường tròn đi qua các đỉnh $A$, $B$, $C$ nên ta có hệ phương trình 
	\begin{eqnarray*}
		& &\heva{&0^2+6^2-2a\cdot 0-2b\cdot 6+c=0\\&7^2+7^2-2a\cdot 7-2b\cdot 7+c=0\\&8^2+0^2-2a\cdot 8-2b\cdot 0+c=0}\\
		&\Leftrightarrow& \heva{&12b-c=36\\&14a+14b-c=98\\&16a-c=64}\\
		&\Leftrightarrow& \heva{&a=4\\&b=3\\&c=0.}
	\end{eqnarray*}
	Vậy đường tròn ngoại tiếp tam giác $ABC$ có phương trình là $x^2+y^2-8x-6y=0$.
}
\end{bt}

		Phương trình tham số đường thẳng $d$ đi qua điểm $B(3;3)$ và có véc-tơ chỉ phương $\vec{u}=(2;5)$ là

\begin{bt}%[0T7K3-2]%[Dự án đề kiểm tra GHK1 22-23-Nhật Thiện]%[THPT Nguyễn Công Trứ]
	Một tam giác vuông có một cạnh góc vuông ngắn hơn cạnh huyền $8$ cm. Tính độ dài của cạnh huyền, biết chu vi tam giác bằng $30$ cm.
	\loigiai{
	Gọi $x$ cm là độ dài một cạnh góc vuông ($x>0$).\\
	Độ dài cạnh huyền bằng $x+8$ cm.\\
	Độ dài cạnh góc vuông còn lại $\sqrt{(x+8)^2-x^2}$ hay $\sqrt{16x+64}$ cm.\\
	Chu vi tam giắc bằng $30$ cm, tức là
	\begin{eqnarray*}
		\sqrt{16x+64}+x+x+8=30&\Leftrightarrow &\sqrt{16x+64}=22-2x\\
		&\Rightarrow & 16x+64=(22-2x)^2\\
		&\Rightarrow & \hoac{&x=5\\&x=21.}
	\end{eqnarray*}
Thử lại, ta thấy $x=5$ thỏa mãn.\\
Vậy độ dài của cạnh huyền bằng $5+8=13$ cm.
}
\end{bt}
