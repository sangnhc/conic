\begin{bt}%[0T2Y2-1]%[0D4Y4-1]%[Dự án đề kiểm tra HKI NH22-23 - Quan Ón]%[Lê Thánh Tông]
	Cho bất phương trình bậc nhất hai ẩn $ 2x-3y+5\leq 0 $	và $ x+2y-3\geq 0 $. Cặp số nào sau đây thỏa mãn đồng thời đồng thời cả hai bất phương trình đã cho? Vì sao?
	\begin{listEX}[2]
		\item $ (1;3) $.
		\item $ (-2;2) $.
	\end{listEX}
	\loigiai{
		\begin{enumerate}
			\item Xét cặp số $ (1;3) $. Ta có $ 2\cdot 1-3\cdot 3+5\leq 0 \Leftrightarrow -2 \leq 0$ (đúng)	và $ 1+2\cdot 3-3\geq 0 \Leftrightarrow 4 \geq 0$ (đúng). 
			\item Xét cặp số $ (-2;2) $. Ta có $ 2\cdot (-2)-3\cdot 2+5\leq 0 \Leftrightarrow -5 \leq 0$ (đúng) và $ -2+2\cdot 2-3\geq 0 \Leftrightarrow -1 \geq 0$ (sai). 
		\end{enumerate}	
		Vậy cặp số $ (1;3) $ thỏa mãn đồng thời đồng thời cả hai bất phương trình đã cho.
	}
\end{bt}

\begin{bt}%[0T5B4-1]%[0H2B2-1]%[Dự án đề kiểm tra HKI NH22-23 - Quan Ón]%[Lê Thánh Tông]
	Cho hình bình hành $ ABCD $	có $ \widehat{ABC}=120^\circ $ và $ AB=2 $; $ AD=4 $. Tính tích vô hướng $ \overrightarrow{AB}\cdot \overrightarrow{AD}$.
	\loigiai{
		\begin{center}
			\begin{tikzpicture}[declare function={r=3;}]
				\path (0,0) coordinate (A)
				(2,0) coordinate (M)

				($ (M)!-1!(A) $)coordinate (D)
				($ (B)+(D)-(A) $)coordinate (C)
				;
				\path (A)--(B)--([turn]90:0.35) coordinate (xt)
				($(A)-(B)+(xt)$) coordinate (yt);
				\draw[dash pattern=on 2pt off 2pt] (A)--(yt) (B)--(xt);
				\draw[>=stealth,|<->|] (xt)--(yt) node[fill=white,inner sep=0pt,font=\scriptsize,midway,sloped]{$\,\,2\ \mathrm{}$};
				\path (A)--(D)--([turn]-90:0.35) coordinate (xt)
				($(A)-(D)+(xt)$) coordinate (yt);
				\draw[dash pattern=on 2pt off 2pt] (A)--(yt) (D)--(xt);
				\draw[>=stealth,|<->|] (xt)--(yt) node[fill=white,inner sep=0pt,font=\scriptsize,midway,sloped]{$\,\,4\ \mathrm{}$};

				}	
			\end{tikzpicture}

		Ta có $ \widehat{ABC}=120^\circ \Rightarrow \widehat{BAC}=60^\circ $.\\
		$ \overrightarrow{AB}\cdot \overrightarrow{AD}=\left\vert \overrightarrow{AB}\right\vert\cdot \left\vert\overrightarrow{AD}\right\vert\cdot \cos \left( \overrightarrow{AB}, \overrightarrow{AD}\right)=2\cdot 4\cdot \cos 60^\circ=4$.
	}
\end{bt}

		\end{eqnarray*}
	}

\begin{bt}%[0T3K1-2]%[0D2K1-2]%[Dự án đề kiểm tra HKI NH22-23 - Quan Ón]%[Lê Thánh Tông]
	\hfill
	\begin{enumerate}
		\item Tìm tập xác định của hàm số $ y=f(x)=x^2+\dfrac{\sqrt{2}}{x^2-1} $.
		\item Tìm các giá trị của tham số $ m $ để tập xác định của hàm số $ y=\sqrt{x-1}+\dfrac{1}{\sqrt{m-x}} $ chứa đúng $ 5 $ số nguyên.
	\end{enumerate}	
	\loigiai{
		\begin{enumerate}
			\item Điều kiện $ x^2-1\neq 0\Leftrightarrow \heva{&x\neq 1\\&x\neq -1.} $\\
			Tập xác định của hàm số là $ \mathscr{D}=\mathbb{R}\setminus\{-1;1\} $.
			\item Điều kiện $ \heva{&x-1\geq 1\\&m-x>0}\Leftrightarrow\heva{&x\geq 1\\&x<m}\Leftrightarrow 1\leq x<m $.\\
			Tập xác định của hàm số là $ \mathscr{D}=[1;m)$.\\
			Tập xác định của hàm số  chứa đúng $ 5 $ số nguyên ta có $ \{1;2;3;4;5\}\subset  \mathscr{D}=[1;m)\Leftrightarrow m=6$.
		\end{enumerate}		
	}
\end{bt}

		\begin{enumerate}

	}

\begin{bt}%[0T4B2-3]%[0H2B3-1]%[Dự án đề kiểm tra HKI NH22-23 - Quan Ón]%[Lê Thánh Tông]
	\hfill
	\begin{enumerate}
		\item Cho tam giác $ ABC $ có $ AB=3a $, $ AC=5a $, $ BC=7a $. Chứng minh tam giác $ ABC $ có một góc tù.
		\item Cho tam giác $ ABC $ có $ \widehat{ABC}=108^\circ $, $ \widehat{ACB}=58^\circ $, $ BC=42$. Tính chiều cao hạ từ $ A $ của tam giác $ ABC $.
	\end{enumerate}
	\loigiai{
			\immini{
			\begin{enumerate}
				\item Ta có $ AB< AC< BC\Rightarrow C<B<A $.\\
				Theo hệ quả định lí cosin ta có\\
				$ \cos A = \dfrac{AB^2+AC^2-BC^2}{2\cdot AB\cdot AC}=\dfrac{(3a)^2+(5a)^2-(7a)^2}{2\cdot 3a\cdot 5a}=-\dfrac{1}{2}$.\\
				Vì $\cos A = -\dfrac{1}{2} < 0 \Rightarrow A$ tù.
				\item Tính chiều cao hạ từ $ A $ của $ \triangle ABC $.
				Áp dụng định lí $ \sin $ trong tam giác $ ABC $, ta có\\
				$ \dfrac{AC}{\sin 108^\circ}= \dfrac{BC}{\sin 14^\circ}\Rightarrow AC= \dfrac{BC\cdot \sin 108^\circ}{\sin 14^\circ}= \dfrac{42\cdot \sin 108^\circ}{\sin 14^\circ}$.\\
				Vì tam giác $ ABC $ có $ \widehat{ABC}=108^\circ $ nên $ H $ nằm ngoài đoạn $ BC $.\\
				Tam giác $ AHC $ vuông tai $ H $ có \\
				$ AH=AC\sin 58^\circ= \dfrac{42\cdot \sin 108^\circ}{\sin 14^\circ}\cdot \sin 58^\circ\approx140$.
			\end{enumerate}	
			}
			{
				\begin{tikzpicture}[scale=0.8, font=\footnotesize, line join=round, line cap=round, >=stealth]
					\path (0,0) coordinate (B)
