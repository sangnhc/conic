		Cho tam giác $ABC$ có $AB=9$, $AC=8$, $BC=7$. Tính diện tích tam giác $ABC$, đường cao $AH$, bán kính đường tròn ngoại tiếp và bán kính đường tròn nội tiếp của tam giác $ABC$.
		%\dapso{$S_{ABC}=12\sqrt{5}$; $AH=\dfrac{24\sqrt{5}}{7}$; $R=\dfrac{21\sqrt{5}}{10}$; $r=\sqrt{5}$}
		\loigiai{
			\begin{itemize}
				\item Ta có $p=\dfrac{AB+AC+BC}{2}=\dfrac{9+8+7}{2}=12$.
				\item Diện tích tam giác $ABC$ là $$S_{ABC}=\sqrt{p(p-AB)(p-AC)(p-BC)}=\sqrt{12\cdot 3\cdot 4\cdot 5}=12\sqrt{5}$$.
				\item $AH=\dfrac{2S_{ABC}}{BC}=\dfrac{24\sqrt{5}}{7}$.
				\item $R=\dfrac{AB\cdot AC\cdot BC}{4S_{ABC}}=\dfrac{9\cdot 8\cdot 7}{4\cdot 12\sqrt{5}}=\dfrac{21\sqrt{5}}{10}$.
				\item $r=\dfrac{S_{ABC}}{p}=\dfrac{12\sqrt{5}}{12}=\sqrt{5}$.
			\end{itemize}
		}
	\end{bt}
