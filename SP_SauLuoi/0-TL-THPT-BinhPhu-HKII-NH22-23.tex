	\begin{enumerate}
		\item Giải phương trình $\sqrt{4x+9}=x+2$.
		\item Giải bất phương trình $(x^2-x-2)(x^2+x)<0$.
		\item Tìm các giá trị của $m$ để $f(x)=mx^2-2mx-3\le 0$, $\forall x\in \mathbb{R}$.
	\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item Ta có
		\begin{eqnarray*}
			& &\sqrt{4x+9}=x+2\\
			&\Rightarrow& 4x+9=(x+2)^2\\
			&\Rightarrow& 4x+9=x^2+4x+4\\
			&\Rightarrow& -x^2+5=0\\
			&\Rightarrow& \hoac{&x=\sqrt{5}\\&x=-\sqrt{5}.}
		\end{eqnarray*}
	Thử lại, ta thấy $x=\sqrt{5}$ thỏa mãn phương trình.\\
	Vậy tập nghiệm của phương trình là $S=\{\sqrt{5}\}$.
	\item 
	Xét biểu thức $f(x)=(x^2-x-2)(x^2+x)$.

	\item
	Xét $m=0$, suy ra $f(x)=-3\le 0$, $\forall x\in \mathbb{R}$ luôn đúng.\\
	Suy ra nhận $m=0$.\\

	\begin{eqnarray*}

	\begin{enumerate}
		\item Có bao nhiêu cách sắp xếp tùy ý $7$ bạn nữ và $3$ bạn nam vào một ghế dài có $10$ chỗ ngồi.
		\item Một hộp có $4$ viên bi đỏ, $5$ viên bi màu xanh, $6$ viên bi màu vàng. Hỏi có bao nhiêu cách lấy $3$ viên bi từ hộp đó mà có đủ $3$ màu?
		\item Từ các chữ số $0$; $1$; $2$; $3$; $4$; $5$; $6$ có thể lập được baoo nhiêu số tự nhiên chẵn có $5$ chữ số khác nhau trong đó có đúng hai số lẻ và hai số lẻ đứng cạnh nhau.
	\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item Xếp tùy ý $7$ bạn nữ và $3$ bạn nam vào một ghế dài có $10!=3628800$ cách.
		\item Chọn $1$  viên bi đỏ, $1$ viên bi xanh, $1$ viên bi vàng có $\mathrm{C}_4^1\cdot \mathrm{C}_5^1\cdot \mathrm{C}_6^1=120$ cách lấy thỏa mãn đề bài.
		\item 
		\begin{itemize}
			\item Xét số có $5$ chữ số khác nhau, trong đó có đúng hai số lẻ và hai số lẻ đứng cạnh nhau.\\
			Chọn $2$ chữ số lẻ có $\mathrm{C}_3^2$ cách. Xếp $2$ chữ số lẻ đứng cạnh nhau có $2!$ cách.\\
			Chọn $3$ số chẵn tùy ý có $\mathrm{C}_4^3$ cách.\\
			Xếp $3$ số chẵn và nhóm $2$ số lẻ có $4!$ cách.\\
			Suy ra có $\mathrm{C}_3^2\cdot 2!\cdot \mathrm{C}_4^3\cdot 4!=576$ cách.
			\item Xét số có $5$ chữ số khác nhau mà đứng đầu là chữ số $0$, trong đó có đúng hai số lẻ và hai số lẻ đứng cạnh nhau.\\
			Chọn $2$ chữ số lẻ có $\mathrm{C}_3^2$ cách. Xếp $2$ chữ số lẻ đứng cạnh nhau có $2!$ cách.\\
			Chọn $2$ số chẵn tùy ý có $\mathrm{C}_4^2$ cách.\\
			Xếp $2$ số chẵn và nhóm $2$ số lẻ có $3!$ cách.\\
			Suy ra có $\mathrm{C}_3^2\cdot 2!\cdot \mathrm{C}_4^2\cdot 3!=216$ cách.
		\end{itemize}
		Vậy có $576-216=360$ số được tạo thành thỏa mãn yêu cầu đề bài.
	\end{enumerate}
}

		\item Viết phương trình tiếp tuyến của đường tròn $(C)$ biết tiếp tuyến song song với đường thẳng $d\colon 3 x-y+1=0$.
		\item Viết phương trình đường thẳng qua điểm $A$ và vuông góc với đường thẳng $AB$.
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Đường tròn $(C)$ có tâm $I(2;-3)$ và bán kính $R=\sqrt{10}$.
	\item Vì tiếp tuyến song song với đường thẳng $d\colon 3 x-y+1=0$ nên phương trình tiếp tuyến $\Delta$ có dạng $3 x-y+C=0$ với $C\ne 1$.\\
	Do $\Delta$ tiếp xúc với đường tròn $(C)$ nên 
	\allowdisplaybreaks
	\begin{eqnarray*}
		\mathrm{d}(I,\Delta)=R&\Leftrightarrow&\dfrac{|3\cdot 2+3+C|}{\sqrt{10}}=\sqrt{10}\\
		&\Leftrightarrow& \left|9+C\right|=10\\
		&\Leftrightarrow& \hoac{&9+C=10\\&9+C=-10}\\
		&\Leftrightarrow& \hoac{&C=1 \text{ (loại)}\\&C=-19 \text{ (nhận)}.}
	\end{eqnarray*} 
Vậy phương trình tiếp tuyến $\Delta$ là $3 x-y-19=0$.
	\item Đường thẳng $(d')\heva{&\text{qua điểm } A(-1 ; 3)\\&\text{vuông góc với đường thẳng } AB\Rightarrow \text{ véc-tơ pháp tuyến }\vec{n}=\overrightarrow{AB}=(2;-1).}$\\
	Suy ra phương trình tổng quát của đường thẳng $(d')\colon 2x-y+5=0$.
\end{enumerate}
}

	\loigiai{
	\begin{enumerate}
		\item Elip có một đỉnh là $(5 ; 0)$ suy ra $a=5$.\\
		Tiêu cự là $4$ suy ra $c=2$ khi đó $b=\sqrt{a^2-b^2}=\sqrt{21}$.\\

		Ta có $\heva{&MA\cdot MB=MI^2-R^2=1\\&MA^2+MB^2=18}\Rightarrow \left(MA-MB\right)^2=16\Rightarrow AB=4$.\\
		Phương trình đường thẳng có dạng $\Delta\colon ax+by-2a-3b=0 \text{ với }\left(a^2+b^2>0\right)$.\\

			&\Leftrightarrow& 2 a^2+3 a b-2 b^2=0 \Leftrightarrow \hoac{&a=-2 b\\&b=2 a.}
		\end{eqnarray*}
