\begin{bt}%[0T7B2-1]%[Dự án đề kiểm tra GHKII NH22-23- Phạm Văn Long]%[THPT Nguyễn An Ninh]
	Giải các bất phương trình sau:
	\begin{listEX}[2]
		\item $x^2-x>9x-30$.
		\item $9x(x+3) \leq-3x-25$.
	\end{listEX}
	\loigiai{
\begin{enumerate}
	\item $x^2-x>9x-30\Leftrightarrow x^2-10x+30>0\Leftrightarrow (x-5)^2+5>0 \, \forall x\in \mathbb{R}$.\\
	Vậy tập nghiệm của bất phương trình là $S=\mathbb{R}$.
	\item $9x(x+3) \leq-3x-25\Leftrightarrow 9x^2+30x+25\leq 0 \Leftrightarrow (3x+5)^2\le 0 \Leftrightarrow x=-\dfrac{5}{3}$.\\
	Vậy tập nghiệm của bất phương trình là $S=\left\{-\dfrac{5}{3}\right\}$.
\end{enumerate}	
}
\end{bt}

	Cho $\vec{a}=(-2;0)$ và $\vec{b}=(1;-5)$
	\begin{enumerate}
		\item Tính $\vec{a}-2 \vec{b}$.
		\item Tính tích vô hướng của hai véc-tơ $\vec{a}$ và $\vec{b}$.
	\end{enumerate}
	\loigiai{
\begin{enumerate}
	\item Ta có $\vec{a}-2 \vec{b}=(-2;0)-2\cdot (1;-5)=(-4;10)$.
	\item Ta có $\vec{a}\cdot \vec{b}=-2\cdot 1+0\cdot (-5)=-2$.
\end{enumerate}	
}

\begin{bt}%[1 điểm]%[0T9Y3-1]%[Dự án đề kiểm tra GHKI NH22-23-Trương Đăng Khoa]%[THPT Nguyễn An Ninh]
	Trong mặt phẳng với hệ tọa độ $Oxy$, đường tròn $(C)$ có phương trình $x^2+y^2-2x+24y-24=0$. Tìm tâm và tính bán kính của đường tròn $(C)$.
	\loigiai{
		Gọi $I(a;b)$ là tâm của đường tròn $(C)$, $R$ là bán kính của đường tròn $(C)$.\\
		Ta có $x^2+y^2-2x+24y-24=0$\\
		Suy ra $\heva{&-2a=-2\\ &-2b=24}\Leftrightarrow\heva{&a=1\\ &b=-12}$\\
		Mặt khác $R=\sqrt{1^2+(-12)^2-(-24)}=13$.\\
		Vậy đường tròn $(C)$ có tâm $I(1;-12)$ và bán kính $R=13$.
	}
\end{bt}

\begin{bt}%[1 điểm]%[0T7B2-1]%[Dự án đề kiểm tra GHKI NH22-23-Trương Đăng Khoa]%[THPT Nguyễn An Ninh]
	Giải phương trình $\sqrt{-2x^2+11x+21}=9-5x$.
	\loigiai{
		Bình phương hai vế phương trình đã cho ta được
		\begin{eqnarray*}
			&&-2x^2+11x+21=(9-5x)^2\\
			&\Leftrightarrow& -2x^2+11x+21=81-90x+25x^2\\
			&\Leftrightarrow& 27x^2-101x+60=0\\
			&\Leftrightarrow& \hoac{&x=3\\ &x=\dfrac{20}{27}.}
		\end{eqnarray*}
		Thay lần lượt các giá trị trên vào phương trình đã cho, ta thấy chỉ có $x=\dfrac{20}{27}$ thỏa mãn.\\
		Vậy nghiệm của phương trình đã cho là $x=\dfrac{20}{27}$.
	}
\end{bt}

	\loigiai{

		Suy ra véc-tơ pháp tuyến $\vv{n}_{AB}=(-14;3)$.\\
		Phương trình tổng quát của đường thẳng $d$ đi qua điểm $A(-3;2)$ và nhận véc-tơ $\vv{n}_{AB}=(-14;3)$ làm véc-tơ pháp tuyến là
		$$-14(x+3)+3(y-2)=0\Leftrightarrow-14x+3y-48=0.$$
	}
