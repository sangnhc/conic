	Xét tính đồng biến, nghịch biến cùa hàm số $f(x)=\dfrac{-3}{2 x-10}$ trên khoảng $(5 ;+\infty)$.
	\loigiai{$\forall x_1 ; x_2\in(5 ;+\infty)$, giả sử $x_1<x_2\Leftrightarrow x_1-x_2<0$.\\
	Ta có 	$f\left(x_1\right)-f\left(x_2\right)=\dfrac{-3}{2x_1-10}+\dfrac 3{2x_2-10}=\dfrac{6\left(x_1-x_2\right)}{\left(2x_1-10\right)\cdot\left(2x_2-10\right)}$.\\

	Do đó $f(x)_1-f\left(x_2\right)<0$.\\
	Vậy hàm số đã cho đồng biến trên khoảng $(5 ;+\infty)$.
	}
\end{bt}

\begin{bt}%[Dự án đề kiểm tra HKII-22-23-Nguyễn Tài Tuệ]%[0T3K2-5]
	Cầu \textit{Ponte Luis} ở Porto (Bồ Đào Nha) có hình dạng đường cong parabol có bề lõm quay xuống dưới để giúp giảm lực mà cây cầu phải gánh. Giả sử ta lập một hệ trục tọa độ $Oxy$ sao cho một chân cầu đi qua gốc tọa độ $O$ như hình vẽ $(x, y$ tính bằng mét), chân kia của cầu ở vị trí $(6 ; 0)$. Biết một điểm $A$ trên cầu có tọa độ là $(2; 4)$. Tìm hàm số bậc hai có đồ thị chứa cung parabol nói trên.
	\begin{center}
			\begin{tikzpicture}[scale=1, font=\footnotesize, line join=round, line cap=round, >=stealth]
			\def\xmin{-1}\def\xmax{6.5}\def\ymin{-1}\def\ymax{5}
			\draw[->] (\xmin-0.2,0)--(\xmax+0.2,0) node[below] {\footnotesize $x$};
			\draw[->] (0,\ymin-0.2)--(0,\ymax+0.2) node[right] {\footnotesize $y$};
			\draw (0,0) node [below left] {\footnotesize $O$};
			\foreach \x in {-1,1,2,3,4,5,6}\draw (\x,0.1)--(\x,-0.1) node [below] {\footnotesize $\x$};
			\foreach \y in {-1,1,2,3,4,5}\draw (0.1,\y)--(-0.1,\y) node [left] {\footnotesize $\y$};
			\clip (\xmin,\ymin) rectangle (\xmax,\ymax);
			\draw[smooth,samples=200,domain=0:6] plot (\x,{-0.5*((\x)^2)+3*\x+0});
			\draw[dashed] (2,0)--(2,4)node [above left] {\footnotesize $A$}--(0,4);	\fill (2,4) circle (1pt);	
		\end{tikzpicture}
	\end{center}
	
\loigiai{
	Gọi hàm số bậc hai $y=ax^2+bx+c$ có đồ thị chứa cung parabol nói trên.\\
	Dựa vào đồ thị ta thấy hàm số đi qua các điểm $O(0;0)$, $A(2;4)$ và $(6;0)$ nên ta có\\
	$$\heva{&0=a\cdot 0^2+b\cdot 0+c\\& 4=a\cdot 2^2+2\cdot b+c\\&0=a\cdot 6^2+b\cdot6+c}\Leftrightarrow \heva{&c=0\\&b=3\\&a=-\dfrac{1}{2}.}$$\\
	Vậy $y=-\dfrac{1}{2}x^2+3x$ là hàm số bậc hai có đồ thị chứa cung parabol nói trên.
	}
\end{bt}





\begin{bt}%[0T6B4-3] %[Dự Án Đề Thi HK1-K10-2022-2023]%[TheHung Nguyen]
Để biết hai loại hoa hồng đỏ (HĐ) và hoa hồng trắng (HT) phát triển như thế nào sau khi gieo hạt, bạn Mai gieo $6$ hạt hoa hồng đỏ và $6$ hạt hoa hồng trắng vào $12$ chậu riêng biệt và cung cấp cho chúng lượng nước, ánh sáng như nhau. Sau $1$ tuần, $12$ hạt đã nảy mầm và phát triển thành $12$ cây con. Bạn Mai đo chiều cao từ rễ đến ngọn của mỗi cây (đơn vị mm) và ghi kết quả là mẫu số liệu sau:
\begin{center}
	\begin{tabular}{|c|c|c|c|c|c|c|}
		\hline
		Cây& HĐ1 & HĐ2 & HĐ3 & HĐ4 & HĐ5 & HĐ6 \\ 
		\hline
	Chiều cao	& $110$  & $101$  & $93$ & $103$ & $104$ & $97$ \\ 
		\hline
	Cây	& HT1 & HT2 & HT3 & HT4 & HT5 & HT6 \\ 
		\hline
	Chiều cao	& $85$ & $90$ & $87$ & $92$ &$100$  & $105$ \\ 
		\hline
	\end{tabular}
\end{center}
	\begin{enumerate}
		\item Hãy tính phương sai của hai mẫu số liệu lấy từ hoa hồng đỏ và hoa hồng trắng? (làm tròn đến hàng phần chục)
		\item Theo em, cây hoa hồng loại nào phát triển đồng đều hơn? Vì sao?
	\end{enumerate}
	\loigiai{	\begin{enumerate}
			\item Chiều cao trung bình của hoa hồng đỏ là $$\overline{x}_\text{Đ}=\dfrac{110+101+93+103+104+97}{6}=\dfrac{304}{3}.$$
			Phương sai của HĐ là
			\begin{eqnarray*}
				& & S^2_\text{Đ}=\dfrac{(110-\overline{x}_\text{Đ})^2+(101-\overline{x}_\text{Đ})^2+(93-\overline{x}_\text{Đ})^2+(103-\overline{x}_\text{Đ})^2}{6}\\
				&& \,\,\,\,\,\,\,\,\,+\dfrac{(104-\overline{x}_\text{Đ})^2+(97-\overline{x}_\text{Đ})^2}{6}\\
				&\Leftrightarrow & S^2_\text{Đ}=\approx 28{,}9.
			\end{eqnarray*}			
			Chiều cao trung bình của hoa hồng trắng là 
			$$\overline{x}_T=\dfrac{85+90+87+92+100+105}{6}=\dfrac{559}{6}.$$
			Phương sai của HT là
			\begin{eqnarray*}
				& & S^2_T=\dfrac{(85-\overline{x}_T)^2+(90-\overline{x}_T)^2+(87-\overline{x}_T)^2+(92-\overline{x}_T)^2}{6}\\
				& &\,\,\,\,\,\,\,\,\,+ \dfrac{(100-\overline{x}_T)^2+(105-\overline{x}_T)^2}{6}\\
				&\Leftrightarrow & S^2_T\approx50{,}5.
			\end{eqnarray*}		
			\item Hoa hồng đỏ phát triển đồng đều hơn hoa hồng trắng.\\
			Vì phương sai của hoa hồng đỏ bé hơn phương sai hoa hồng trắng.
		\end{enumerate}
		
	}
\end{bt}
