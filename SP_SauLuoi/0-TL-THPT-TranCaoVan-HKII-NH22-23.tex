	\begin{enumerate}
		\item Tính xác suất $2$ quả bóng lấy ra có cùng màu.
		\item Hỏi trong hộp có bao nhiêu quả bóng? 
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Gọi $A$ là biến cố \lq\lq hai quả bóng lấy ra có cùng màu\rq\rq. Khi đó $\overline{A}$ là biến cố \lq\lq hai quả bóng lấy ra khác màu\rq\rq.\\
			Theo đề, ta có $P(\overline{A})=\dfrac{10}{21}$. Do đó
			\begin{eqnarray*}
				P(A)=1-P(\overline{A})=1-\dfrac{10}{21}=\dfrac{11}{21}.
			\end{eqnarray*}
			\item Gọi $n$ là số quả bóng trắng có trong hộp $(n \in \mathbb{N^*})$.\\
			Biến cố $B$: \lq\lq Chọn được $2$ quả bóng khác màu\rq\rq.\\
			Số cách chọn $2$ quả bóng khác màu là $n(A)=2n$ (cách).\\
			Số cách chọn $2$ quả bóng bất kỳ trong $(n+2)$ quả bóng là 
			\begin{center}
				$\mathrm{C}_{n+2}^2 = \dfrac{(n+2)!}{n!\cdot 2!} = \dfrac{(n+2)(n+1)}{2}$ (cách).
			\end{center}
			Xác suất của biến cố $B$ là \begin{center}
				$\mathrm{P}(B)=\dfrac{n(B)}{n(\Omega)} = \dfrac{2n}{\dfrac{(n+2)(n+1)}{2}} = \dfrac{4n}{n^2+3n+2}$.
			\end{center}
			Khi đó, ta có 
			\allowdisplaybreaks
			\begin{eqnarray*}
			\dfrac{4n}{n^2+3n+2} = \dfrac{10}{21}
				&\Rightarrow& 84n = 10(n^2+3n+2)\\
				&\Rightarrow&  10n^2 - 54n +20 =0 \\
				&\Rightarrow& \hoac{& n=5 \text{ (nhận)} &\\&n= \dfrac{2}{5} \text{ (loại)}.} 
			\end{eqnarray*}
			Vậy trong hộp có tất cả $7$ quả bóng.
		\end{enumerate}
	}
