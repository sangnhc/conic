\begin{bt}%[0T9B2-6]%[Dự án đề kiểm tra GHKII NH22-23- Thầy Hóa]%[THPT Tây Thạnh]
Trong mặt phẳng $Oxy$, cho tam giác $ABC$ có tọa độ các đỉnh là $A(2;3)$, $B(4;5)$ và $C(6;5)$.
\begin{enumerate}
	\item Tính độ dài đoạn thẳng $AB$ và số đo góc $B$ trong tam giác $ABC$.
	\item Tìm tọa độ điểm $D\left(m;m^2-6m-4\right)$ sao cho tứ giác $ABCD$ là hình thang có hai đáy là $BC$, $AD$.
	\item Lập phương trình tham số của trung tuyến $AM$.
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Ta có $\begin{aligned}[t]
		&\vec{AB}=(2;2)\Rightarrow AB=\sqrt{2^2+2^2}=2\sqrt{2};\\
		&\vec{AC}=(4;2)\Rightarrow AC=\sqrt{4^2+2^2}=2\sqrt{5};\\
		&\vec{BC}=(2;0)\Rightarrow BC=\sqrt{2^2+0^2}=2.
	\end{aligned}$\\
	Áp dụng định lý cô-sin, ta có
	\allowdisplaybreaks
	\begin{eqnarray*}
		\cos B&=&\dfrac{BC^2+AB^2-AC^2}{2\cdot BC\cdot AB}\\
		&=&\dfrac{4+8-20}{2\cdot 2\cdot 2\sqrt{2}}\\
		&=&\dfrac{-1}{\sqrt{2}}.
	\end{eqnarray*}
	Suy ra $\widehat{B}=135^\circ$.
	\item Ta có $\vec{AD}=\left(m-2;m^2-6m-7\right)$.\\
	Vì $ABCD$ là hình thang có $BC\parallel AD$ nên $\vec{AD}=k\vec{BC}\ (k\neq 0)$. Do đó
	\[m^2-6m-7=0\Leftrightarrow \hoac{&m=-1\\&m=7.} \]
	\begin{itemize}
		\item Với $m=-1\Rightarrow D(-1;3)$.
		\item Với $m=7\Rightarrow D(7;3)$.
	\end{itemize}
	Vậy có 2 điểm $D$ thỏa mãn yêu cầu bài toán là $D(-1;3)$ và $D(7;3)$.
	\item Vì $M$ là trung điểm của $BC$ nên $M(5;5)$.\\
	Ta có $\vec{AM}=(3;2)$. Phương trình tham số của đường thẳng $AM$
	\[AM\colon\heva{&x=2+3t\\&y=3+2t}\ (t\in\mathbb{R}). \]
\end{enumerate}
}
\end{bt}

\begin{bt}%[0T9B2-2]%[Dự án đề kiểm tra GHKII NH22-23- Thầy Hóa]%[THPT Tây Thạnh]
Trong mặt phẳng $Oxy$, cho đường thẳng $\Delta$ có phương trình $\heva{&x=4+t\\&y=4+2t}$ và điểm $E(5;11)$.
\begin{enumerate}
	\item Xác định một véc-tơ chỉ phương của đường thẳng $\Delta$.
	\item Lập phương trình tổng quát của đường thẳng $d_1$ đi qua điểm $E$ và vuông góc với đường thẳng $\Delta$.
	\item Lập phương trình tổng quát đường thẳng $d_2$ song song với đường thẳng $\Delta$ và khoảng cách từ điểm $E$ đến đường thẳng $d_2$ bằng $\sqrt{5}$.
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Một véc-tơ chỉ phương của đường thẳng $\Delta$ là $\vec{u}=(1;2)$.
	\item Vì $d_1$ vuông góc với $\Delta$ nên $d_1$ nhận véc-tơ chỉ phương $\vec{u}=(1;2)$ của $\Delta$ làm véc-tơ pháp tuyến.\\
	Phương trình tổng quát của $d_1$:
	\[(d_1):1(x-5)+2(y-11)=0\Leftrightarrow x+2y-27=0. \]
	\item Đường thẳng $\Delta$ đi qua điểm $(4;4)$ và có véc-tơ chỉ phương $\vec{u}=(1;2)$, suy ra véc-tơ pháp tuyến là $\vec{n}=(2;-1)$. Phương trình tổng quát của $\Delta$
	\[(\Delta):2\cdot (x-4)-1\cdot (y-4)=0\Leftrightarrow 2x-y-4=0. \]
	Vì $d_2$ song song với $\Delta$ nên phương trình tổng quát của $d_2$ có dạng:
	\[(d_2):2x-y+c=0\ (c\neq -4). \]
	Vì khoảng cách từ $E$ đến $d_2$ bằng $\sqrt{5}$ nên
	\allowdisplaybreaks
	\begin{eqnarray*}
	\mathrm{d}\left(E,d_2\right)=\sqrt{5}&\Leftrightarrow &\dfrac{\left|10-11+c\right|}{\sqrt{2^2+(-1)^2}}=\sqrt{5}\\
	&\Leftrightarrow &\left|c-1\right|=5\\
	&\Leftrightarrow &\hoac{&c-1=5\\&c-1=-5}\Leftrightarrow\hoac{&c=6\ (\text{nhận})\\&c=-4\ (\text{loại}).}
	\end{eqnarray*}
	Vậy đường thẳng $d_2$ thỏa mãn yêu cầu bài toán là
	\[\left(d_2\right):2x-y+6=0. \]
\end{enumerate}
}
\end{bt}

\begin{bt}%[0T3K2-5]%[Dự án đề kiểm tra GHKII NH22-23- Thầy Hóa]%[THPT Tây Thạnh]
Tại một hộ gia đình nuôi các diêu hồng, người ta cho cá ăn $x$ (gram) thức ăn trên một đơn vị diện tích ở mỗi lần cho ăn. Theo số liệu nghiên cứu được thì đến khi thu hoạch thì mỗi con cá sẽ có cân nặng trung bình tính bởi hàm số $h(x)=-0{,}49x^2+31{,}5x$ (gram). Hỏi hộ gia đình cần cho cá ăn lượng thức ăn bao nhiêu để khi thu hoạch mỗi con cá có cân nặng tối thiểu $500$ (gram).
\loigiai{
Xét bất phương trình: $\begin{aligned}[t]
	&h(x)\geq 500\\
	\Rightarrow&-0{,}49x^2+31{,}5x\geq 500\\
	\Leftrightarrow&-0{,}49x^2+31{,}5x-500\geq 0.
\end{aligned}$\\
Ta có $-0{,}49x^2+31{,}5x-500=0\Leftrightarrow \hoac{&x=\dfrac{200}{7}\\&x=\dfrac{250}{7}.}$\\
Bảng xét dấu
\begin{center}
	\begin{tikzpicture}
		\tkzTabInit[nocadre=false,lgt=2,espcl=2,deltacl=0.5]{$x$/1 ,$VT$/1}
		{$-\infty$ , $\tfrac{200}{7}$ , $\tfrac{250}{7}$ , $+\infty$}
		\tkzTabLine{  , - , 0 , + , 0 , - , }
	\end{tikzpicture}
\end{center}
Dựa vào bảng xét dấu thì
\[-0{,}49x^2+31{,}5x-500\geq 0\Leftrightarrow \dfrac{200}{7}\leq x\leq \dfrac{250}{7}. \]
Vậy để khi thu hoạch mỗi con cá có cân nặng tối thiểu $500$ (gram) thì hộ gia đình cần cho cá ăn lượng thức ăn từ $\dfrac{200}{7}$ (gram) đến $\dfrac{250}{7}$ (gram).
}
\end{bt}
