\begin{ex}%[0H4K3-1]%[Dự án đề kiểm tra GHKI NH22-23- VU Ngoc Hao]%[THPT Lý Thường Kiệt]
	\immini{Khoảng cách từ nhà An ở vị trí $A$ đến nhà Bình ở vị trí $B$ là $500$ m. Từ nhà Bình đi $x$ mét theo phương tạo với $AB$ một góc $60^\circ$ thì sẽ đến nhà bác Mai ở vị trí $M$  và nếu đi thêm $300$ m nữa thì sẽ đến siêu thị ở vị trí $S$ (hình vẽ). Biết rằng quãng đường từ nhà An đến siêu thị bằng $\dfrac{7}{5}$ quãng đường từ nhà An đến nhà bác Mai. Khi đó quãng đường từ nhà Bình đến nhà bác Mai là
	\choice
	{\True $625$ m}

	{$600$ m}
	{$400$ m}}{\begin{tikzpicture}[scale=0.7]
		\path

		(B)++(60:4.5) coordinate (M)

		;

		($(\g:3mm)+(\x)$) node {$\x$};
		\path
		(M)--(B)node[pos=0.5,left]{$x$}

		$$AM^2 = AB^2 +BM^2 - 2\cdot AB\cdot BM\cdot \cos 60^\circ = 500^2 +x^2 - 2\cdot 500\cdot x\cdot \dfrac{1}{2}= x^2 -500x +250000.$$

\begin{bt}%[0T7B3-2]%[Dự án đề kiểm tra GHKII NH22-23- Dương Phước Sang]%[THPT Lý Thường Kiệt]
		Giải phương trình $\sqrt{2x^2+x+3}=1-x$.
		\loigiai{
			Bình phương hai vế của phương trình $\sqrt{2x^2+x+3}=1-x$, ta được
			$$2x^2+x+3=(1-x)^2 
			\Leftrightarrow x^2+3x+2=0
			\Leftrightarrow \hoac{&x=-1\\&x=-2.}$$
			Thay $x=-1$ vào phương trình ta được $\sqrt{2(-1)^2+(-1)+3}=1-(-1)$ là mệnh đề đúng.\\
			Thay $x=-2$ vào phương trình ta được $\sqrt{2(-2)^2+(-2)+3}=1-(-2)$ là mệnh đề đúng.\\
			Vậy phương trình đã cho có tập nghiệm $S=\{-1;-2\}$.
		}
	\end{bt}

	\begin{bt}%[0T7K2-1]%[Dự án đề kiểm tra GHKII NH22-23- Dương Phước Sang]%[THPT Lý Thường Kiệt]
		Tìm tất cả các tham số $m$ để bất phương trình $(m-1)x^2-2(m+1)x+3m-6 \geq 0$ thỏa mãn với mọi $x \in \mathbb{R}$.
		\loigiai{
			Xét bất phương trình $(m-1)x^2-2(m+1)x+3m-6 \geq 0$.\\
			Với $m=1$, bất phương trình trở thành
			\begin{center}
				$-4x-3 \geq 0 \Leftrightarrow x \leq -\dfrac{3}{4}$ (không thỏa mãn yêu cầu của bài toán).
			\end{center}
			Với $m \neq 1$, yêu cầu của bài toán trở thành $\heva{&a>0\\&\Delta' \leq 0}$
			$$\Leftrightarrow
			\heva{&m-1>0\\&(m+1)^2-(m-1)(3m-6) \leq 0}
			\Leftrightarrow
			\heva{&m>1\\&-2m^2+11m-5 \leq 0.} (*)$$
			Cho $-2m^2+11m-5=0 \Leftrightarrow \hoac{&m=5\\&m=\dfrac{1}{2}.}$\\
			Bảng xét dấu
			\begin{center}
				\begin{tikzpicture}
					\tkzTabInit[nocadre=true,lgt=3.4,espcl=2,deltacl=0.6]
					{$m$/1,$-2m^2+11m-5$/1}
					{$-\infty$,$\dfrac{1}{2}$,$5$,$+\infty$}
					\tkzTabLine{,-,0,+,0,-,}
				\end{tikzpicture}
			\end{center}
			Như vậy $(*) \Leftrightarrow \heva{&m>1\\&m \leq \dfrac{1}{2} \vee m \geq 5} \Leftrightarrow m \geq 5$.
		}
	\end{bt}

	\begin{bt}%[0T9K2-6]%[Dự án đề kiểm tra GHKII NH22-23- Dương Phước Sang]%[THPT Lý Thường Kiệt]
		Trong mặt phẳng $Oxy$, cho tam giác $ABC$ có hai đỉnh $A(0;1)$ và $B(3;5)$. Tìm tọa độ đỉnh $C$ thuộc trục tung sao cho diện tích tam giác $ABC$ bằng $6$ (đơn vị diện tích).
		\loigiai{
			Với $A(0;1)$ và $B(3;5)$, ta có $AB=\sqrt{(3-0)^2+(5-1)^2}=5$.\\
			Phương trình của đường thẳng $AB\colon \dfrac{x}{3}=\dfrac{y-1}{4} \Leftrightarrow 4x-3y+3=0$.\\
			Do $C \in Oy$ nên $C(0;m)$ và $\mathrm{d}(C,AB)=\dfrac{|4\cdot 0-3m+3|}{\sqrt{4^2+(-3)^2}}=\dfrac{|3-3m|}{5}$.\\
			Theo giả thiết $S_{\triangle ABC}=6$ nên
			$$\dfrac{1}{2}AB\cdot \mathrm{d}(C,AB)=6
			\Leftrightarrow \dfrac{5}{2}\cdot \dfrac{|3-3m|}{5}=6
			\Leftrightarrow |1-m|=4 
			\Leftrightarrow \hoac{&m=-3\\&m=5.}$$
			Vậy $C(0;-3)$ hoặc $C(0;5)$.
		}
	\end{bt}
