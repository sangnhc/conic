\loigiai{
Đặt $f(x)=-3x^2+x+4$.\\
Ta có 
\begin{itemize}
	\item $a=-3<0$.
	\item $f(x)=0\Leftrightarrow \hoac{&x=-1\\ &x=\dfrac{4}{3}.}$

\end{center}
Vậy tập nghiệm của bất phương trình là $S=\left[-1;\dfrac{4}{3}\right]$.
}

\begin{bt}%[0T7Y3-2]%[Dự án đề kiểm tra HKII NH22-23- Trương Đăng Khoa]%[THPT Lê Trọng Tấn]
Giải phương trình $\sqrt{4x^2+2x+10}=3x+1$.
\loigiai{
Bình phương hai vế phương trình đã cho, ta được
\allowdisplaybreaks
\begin{eqnarray*}
&&4x^2+2x+10=(3x+1)^2\\
&\Leftrightarrow& 4x^2+2x+10=9x^2+6x+1\\
&\Leftrightarrow& 5x^2+4x-9=0\\
&\Leftrightarrow& \hoac{&x=1\\ &x=-\dfrac{9}{5}.}
\end{eqnarray*}
Thay lần lượt các giá trị trên vào phương trình đã cho, ta chỉ thấy có $x=1$ thỏa mãn.\\
Vậy nghiệm của phương trình đã cho là $x=1$.
}
\end{bt}

\begin{bt}%[0T8B3-2]%[Dự án đề kiểm tra HKII NH22-23- Trương Đăng Khoa]%[THPT Lê Trọng Tấn]
Khai triển nhị thức $(2x+1)^4$ và tìm hệ số $x^3$ trong khai triển.
\loigiai{
Ta có $\begin{aligned}[t]
	(2x+1)^4=&\ \mathrm{C}_4^0\cdot (2x)^4 +\mathrm{C}_4^1\cdot (2x)^3\cdot 1^1 +\mathrm{C}_4^2\cdot (2x)^2\cdot 1^2+\mathrm{C}_4^3\cdot (2x)\cdot 1^3+\mathrm{C}_4^4\cdot 1^4\\
	=&\ 16x^4+32x^3+24x^2+8x+1.
\end{aligned}$\\
Vậy hệ số $x^3$ trong khai triển là $32$.
}
\end{bt}

\begin{bt}%[0T8Y2-1]%[Dự án đề kiểm tra HKII NH22-23- Trương Đăng Khoa]%[THPT Lê Trọng Tấn]
Lớp $10$C có $18$ bạn nữ và $24$ bạn nam. Có bao nhiêu cách để
\begin{enumerate}
	\item Chọn ra một bạn để đại diện lớp lên nhận cờ thi đua tuần.
	\item Chọn ra một bạn nam và một bạn nữ để đi dự đại hội.
\end{enumerate}
\loigiai{
Tổng số học sinh lớp $10$C là $18+24=42$.
\begin{enumerate}
	\item Số cách chọn ra một bạn để đại diện lớp lên nhận cờ thi đua tuần trong tổng số $42$ bạn có $\mathrm{C}_{42}^1=42$ cách chọn.
	\item Số cách ra một bạn nam trong $24$ bạn nam để đi dự đại hội có $\mathrm{C}_{24}^1=24$ cách chọn.\\
	Số cách ra một bạn nữ trong $18$ bạn nữ để đi dự đại hội có $\mathrm{C}_{18}^1=18$ cách chọn.\\
	Theo quy tắc nhân ta có $24\cdot 18=432$ cách chọn.
\end{enumerate}
}
\end{bt}

\begin{bt}%[0T8K1-3]%[Dự án đề kiểm tra HKII NH22-23- Trương Đăng Khoa]%[THPT Lê Trọng Tấn]
Trong hộp có $5$ viên bi vàng, $6$ viên bi đỏ và $7$ viên bi xanh. Có bao nhiêu cách chọn ngẫu nhiên $4$ viên bi sao cho số bi vàng bằng với số bi đỏ.
\loigiai{
Chọn ngẫu nhiên $4$ viên bi sao cho số bi vàng bằng với số bi đỏ, ta có các trường hợp sau
\begin{itemize}
	\item \textbf{Trường hợp 1.} Chọn $1$ bi vàng, $1$ bi đỏ và $2$ bi xanh.\\
	Số cách chọn $1$ bi vàng trong $5$ bi vàng có $\mathrm{C}_5^1=5$ cách.\\
	Số cách chọn $1$ bi đỏ trong $6$ bi đỏ có $\mathrm{C}_6^1=6$ cách.\\
	Số cách chọn $2$ bi xanh trong $7$ bi xanh có $\mathrm{C}_7^2=21$ cách.\\
Theo quy tắc nhân ta có $5\cdot 6\cdot 21=630$.
\item \textbf{Trường hợp 2.} Chọn $2$ bi vàng, $2$ bi đỏ.\\
Số cách chọn $2$ bi vàng trong $5$ bi vàng có $\mathrm{C}_5^2=10$ cách.\\
Số cách chọn $2$ bi đỏ trong $6$ bi đỏ có $\mathrm{C}_6^2=15$ cách.\\
Theo quy tắc nhân ta có $10\cdot 15=150$ cách.
\item \textbf{Trường hợp 3.} Chọn $4$ bi xanh.\\
Số cách chọn $4$ bi xanh trong $7$ bi xanh có $\mathrm{C}_7^4=35$ cách.
\end{itemize}
Theo quy tắc cộng, ta có $630+150+35=815$ cách chọn.
}
\end{bt}

\loigiai{

		\item Số tự nhiên có $5$ chữ số khác nhau và chia hết cho $25$ có dạng $\overline{abc25}$, $\overline{abc50}$, $\overline{abc75}$.\\
			Có $7\cdot \mathrm{A}_{7}^{2}\cdot 2$ số dạng $\overline{abc25}$, $\overline{abc75}$.\\

		\item Phép thử $T\colon$ \lq\lq Chọn ngẫu nhiên một số tự nhiên có $5$ chữ số khác nhau\rq\rq.\\

			Có $n(A)=\mathrm{C}_{924}^{1}=924$.
		\item Xác suất cần tìm là $\mathrm{P}(A)=\dfrac{n(A)}{n(\Omega)}=\dfrac{924}{27216}=\dfrac{11}{324}$.

	}

\begin{bt}%[0H9B4-2]%[0H9B2-2]%[0H9B3-2]%[Dự án đề kiểm tra HKII NH22-23 - Don Lee]%[Lê Trọng Tấn]

	\begin{enumerate}[a)]
		\item Viết phương trình chính tắc của $(E)$ có một đỉnh là $A$ và nhận $F$ làm một tiêu điểm.
		\item Viết phương trình tổng quát của đường thẳng đi qua $M$ và vuông góc với đường thẳng $d\colon 2x-y+1=0$.
		\item Viết phương trình đường tròn tâm $F$ và tiếp xúc với đường thẳng $AM$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}[a)]
			\item Gọi phương trình chính tắc của $(E)$ có dạng $\dfrac{x^2}{a^2}+\dfrac{y^2}{b^2}=1$, $(0<b<a)$.\\
				Vì $(E)$ có một đỉnh là $A(-5;0)$ và nhận $F(3;0)$ làm một tiêu điểm nên $a=5$; $c=3$.\\
				Ta có $b^2=a^2-c^2=16$.\\
				Vậy phương trình chính tắc của $(E)$ là $\dfrac{x^2}{25}+\dfrac{y^2}{16}=1$.
			\item Gọi $\Delta$ là đường thẳng đi qua $M$ và vuông góc với đường thẳng $d\colon 2x-y+1=0$.\\
				$\Delta$ có một véc-tơ pháp tuyến $\overrightarrow{n}_{\Delta}=(1;2)$.\\
				Phương trình tổng quát của $\Delta$ là $1\cdot(x-1)+2\cdot(y+2)=0$ hay $\Delta\colon x+2y+3=0$.
			\item Gọi $(C)$ là đường tròn tâm $F$ và tiếp xúc với đường thẳng $AM$, ta có $R=\mathrm{d}\left(F,AM\right)$ là bán kính của $(C)$.\\
				Ta có $\overrightarrow{AM}=(6;-2)$, suy ra $\overrightarrow{n}=(1;3)$ là một véc-tơ pháp tuyến của $AM$.\\
				Phương trình đường thẳng $AM$ là $1\cdot(x+5)+3\cdot(y-0)=0$ hay $AM\colon x+3y+5=0$.\\
				Suy ra $R=\mathrm{d}\left(F,AM\right)=\dfrac{\left|3+3\cdot 0+5\right|}{\sqrt{1^2+3^2}}=\dfrac{8}{\sqrt{10}}$.\\
				Vậy phương trình đường tròn $(C)\colon (x-3)^2+y^2=\dfrac{32}{5}$. 
		\end{enumerate}
	}
