\begin{bt}%[0K3B6-1]%[Dự án đề kiểm tra NH22-23 - Huỳnh Quy]%[THPT Phú Nhuận]
	Cho $\triangle{ABC}$ có $AB=6$, $AC=10$ và $\widehat{A}=120^\circ$. Tính độ dài cạnh $BC$ và đường cao kẻ từ $A$ của $\triangle{ABC}$.
	\dapso{$BC=14$; $h_{a}=\dfrac{15\sqrt{3}}{7}$.}
	\loigiai{ 
	Ta có 
	\begin{eqnarray*}
		BC^2&=&AB^2+AC^2-2\cdot AB\cdot AC\cdot \cos \widehat{A}\\
		&=&6^2+10^2-2\cdot 6\cdot 10\cdot \cos 120^{\circ}\\
		&=&196.\\
		\Rightarrow BC&=&14.
	\end{eqnarray*}
		Lại có
		\begin{eqnarray*}
			S_{\triangle ABC}&=&\dfrac{1}{2}AB\cdot AC\cdot\sin\widehat{A}\\
			&=&\dfrac{1}{2}\cdot 6\cdot 10\cdot \sin120^{\circ}\\
			&=&15\sqrt{3}.
		\end{eqnarray*}
	Mà $S_{\triangle ABC}=\dfrac{1}{2}AH\cdot BC \Rightarrow AH=\dfrac{2S_{\triangle ABC}}{BC}=\dfrac{2\cdot 15\sqrt{3}}{14}=\dfrac{15\sqrt{3}}{7}$.}
\end{bt}

		\item Tính $\overrightarrow{AB}\cdot\overrightarrow{AC}$ theo $a$.
		\item Phân tích $\overrightarrow{BN}$, $\overrightarrow{AM}$ theo các vecto $\overrightarrow{AB}$ và $\overrightarrow{AC}$.
		\item Chứng minh $AM \bot BN$.
	\end{listEX}
\dapso{a) $\dfrac{1}{2}a^2$; b) $\overrightarrow{BN}=-\overrightarrow{AB}+\dfrac{4}{5}\overrightarrow{AC}$, $\overrightarrow{AM}=\dfrac{1}{3}\overrightarrow{AB}+\dfrac{2}{3}\overrightarrow{AC}$.}

			&=&\vec{AB}+\dfrac{2}{3}\vec{BC}\\

			&=&\dfrac{1}{3}\vec{AB}+\dfrac{2}{3}\vec{AC}.
